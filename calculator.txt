<?php

function calculadora($numA, $numB, $op) {
    if ($op == "somar") {
        return $numA + $numB;
    } elseif ($op == "subtrair") {
        return $numA - $numB;
    } elseif ($op == "dividir") {
        return $numA / $numB;
    } elseif ($op == "multiplicar") {
        return $numA * $numB;
    } else {
        return "ESSE OPERADOR NÃƒO EXISTE!";
    }
}

echo calculadora(1, 2, "somar");
